    <!-- Import Three.js core and add-ons from CDN -->

    <script type="importmap">

        {

            "imports": {

                "three": "https://unpkg.com/three@0.128.0/build/three.module.js",

                "three/addons/": "https://unpkg.com/three@0.128.0/examples/jsm/"

    <script type="module">

        import * as THREE from 'three';

        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

        import { CSS2DRenderer, CSS2DObject } from 'three/addons/renderers/CSS2DRenderer.js';


        // --- Setup Scene, Cameras, Renderers ---

        const scene = new THREE.Scene();

        scene.background = new THREE.Color(0x111122);

        scene.fog = new THREE.FogExp2(0x111122, 0.008);


        // Perspective camera

        const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 1000);

        camera.position.set(25, 18, 28);

        camera.lookAt(5, 0, 5);


        // WebGL renderer

        const renderer = new THREE.WebGLRenderer({ antialias: true });

        renderer.setSize(window.innerWidth, window.innerHeight);

        renderer.shadowMap.enabled = true; // enable shadows

        renderer.shadowMap.type = THREE.PCFSoftShadowMap;

        document.body.appendChild(renderer.domElement);


        // CSS2 renderer for text labels

        const labelRenderer = new CSS2DRenderer();

        labelRenderer.setSize(window.innerWidth, window.innerHeight);

        labelRenderer.domElement.style.position = 'absolute';

        labelRenderer.domElement.style.top = '0px';

        labelRenderer.domElement.style.left = '0px';

        labelRenderer.domElement.style.pointerEvents = 'none';

        document.body.appendChild(labelRenderer.domElement);


        // --- Controls ---

        const controls = new OrbitControls(camera, renderer.domElement);

        controls.enableDamping = true;

        controls.dampingFactor = 0.05;

        controls.rotateSpeed = 1.0;

        controls.zoomSpeed = 1.2;

        controls.panSpeed = 0.8;

        controls.target.set(8, 1, 6);


        // --- Lighting ---

        // Ambient light

        const ambientLight = new THREE.AmbientLight(0x404060);

        scene.add(ambientLight);

        // Main directional light (sun)

        const dirLight = new THREE.DirectionalLight(0xffffff, 1);

        dirLight.position.set(10, 20, 5);

        dirLight.castShadow = true;

        dirLight.receiveShadow = true;

        dirLight.shadow.mapSize.width = 1024;

        dirLight.shadow.mapSize.height = 1024;

        dirLight.shadow.camera.near = 0.5;

        dirLight.shadow.camera.far = 40;

        dirLight.shadow.camera.left = -15;

        dirLight.shadow.camera.right = 15;

        dirLight.shadow.camera.top = 15;

        dirLight.shadow.camera.bottom = -15;

        scene.add(dirLight);

        // Fill light from below

        const fillLight = new THREE.PointLight(0x4466cc, 0.3);

        fillLight.position.set(5, 3, 5);

        scene.add(fillLight);

        // Warm back light

        const backLight = new THREE.PointLight(0xffaa66, 0.4);

        backLight.position.set(-3, 5, -5);

        scene.add(backLight);

        

        // Helper: ground grid

        const gridHelper = new THREE.GridHelper(60, 20, 0x88aaff, 0x335588);

        gridHelper.position.y = -0.05;

        gridHelper.material.transparent = true;

        gridHelper.material.opacity = 0.5;

        scene.add(gridHelper);


        // Optional: simple ground plane (transparent but receives shadows)

        const groundPlane = new THREE.Mesh(

            new THREE.PlaneGeometry(45, 35),

            new THREE.ShadowMaterial({ opacity: 0.4, color: 0x000000, transparent: true, side: THREE.DoubleSide })

        );

        groundPlane.rotation.x = -Math.PI / 2;

        groundPlane.position.y = -0.04;

        groundPlane.receiveShadow = true;

        scene.add(groundPlane);


        // --- Helper function to create a labeled room ---

        function addRoom(x, z, width, depth, height, color, name, yOffset = 0) {

            const geometry = new THREE.BoxGeometry(width, height, depth);

            const material = new THREE.MeshStandardMaterial({ color: color, roughness: 0.4, metalness: 0.1, transparent: true, opacity: 0.85 });

            const cube = new THREE.Mesh(geometry, material);

            cube.position.set(x, height/2 + yOffset, z);

            cube.castShadow = true;

            cube.receiveShadow = true;

            scene.add(cube);

            

            // Edges outline

            const edgesGeo = new THREE.EdgesGeometry(geometry);

            const edgesMat = new THREE.LineBasicMaterial({ color: 0xffffff });

            const wireframe = new THREE.LineSegments(edgesGeo, edgesMat);

            wireframe.position.copy(cube.position);

            scene.add(wireframe);

            

            // CSS2D Text label

            const div = document.createElement('div');

            div.textContent = name;

            div.style.color = '#fff';

            div.style.fontSize = '16px';

            div.style.fontWeight = 'bold';

            div.style.textShadow = '1px 1px 0px black';

            div.style.backgroundColor = 'rgba(0,0,0,0.6)';

            div.style.padding = '4px 10px';

            div.style.borderRadius = '20px';

            div.style.border = '1px solid rgba(255,255,255,0.3)';

            div.style.whiteSpace = 'nowrap';

            const label = new CSS2DObject(div);

            label.position.set(x, height + 0.2 + yOffset, z);

            scene.add(label);

            

            return cube;

        }


        function addFlatArea(x, z, width, depth, color, name, yLevel = 0.02) {

            const geometry = new THREE.BoxGeometry(width, 0.05, depth);

            const material = new THREE.MeshStandardMaterial({ color: color, roughness: 0.6, metalness: 0.05 });

            const plate = new THREE.Mesh(geometry, material);

            plate.position.set(x, yLevel, z);

            plate.receiveShadow = true;

            plate.castShadow = false;

            scene.add(plate);

            

            const div = document.createElement('div');

            div.textContent = name;

            div.style.color = '#eee';

            div.style.fontSize = '14px';

            div.style.backgroundColor = 'rgba(0,0,0,0.5)';

            div.style.padding = '2px 8px';

            div.style.borderRadius = '12px';

            const label = new CSS2DObject(div);

            label.position.set(x, 0.25, z);

            scene.add(label);

        }

        

        // --- Define positions (X, Z) ---

        // Badminton Court 1: 14 x 6 (width along X = 14, depth along Z = 6)

        addFlatArea(0, 0, 14, 6, 0x5a7d9a, "🏸 Badminton Court 1\n14 x 6 m", 0.03);

        // Badminton Court 2: 14 x 5.5

        addFlatArea(0, -7.5, 14, 5.5, 0x4e6b8a, "🏸 Badminton Court 2\n14 x 5.5 m", 0.03);

        

        // Meeting Room 1: 200 m² -> approximate 14.14 x 14.14, but to fit layout we make 16 x 12.5 = 200

        addFlatArea(10, 5.5, 16, 12.5, 0x8fa68f, "🪑 Meeting Room 1\n200 m²", 0.04);

        // Meeting Room 2: 50 m² -> 7 x 7.14

        addFlatArea(10, -2, 7, 7.14, 0x7c9e7c, "🪑 Meeting Room 2\n50 m²", 0.04);

        

        // Dance Room 1: 55 m² -> 7.5 x 7.33

        addFlatArea(-7, 6, 7.5, 7.33, 0xb87c7c, "💃 Dance Room 1\n55 m²", 0.04);

        // Dance Room 2: 55 m² -> same size

        addFlatArea(-7, -1.5, 7.5, 7.33, 0xb87c7c, "💃 Dance Room 2\n55 m²", 0.04);

        

        // Office Room (assumed 5x5)

        addFlatArea(-7, -8, 5, 5, 0xc9ae74, "📋 Office Room", 0.04);

        // Reception (assumed 5x4)

        addFlatArea(-1, -8.5, 5, 4, 0xc9ae74, "📋 Reception", 0.04);

        

        // --- Outside Parking for 15 cars (visual representation: 15 small rectangles + asphalt)

        const parkingXStart = -5;

        const parkingZStart = 10.5;

        const carWidth = 0.8;

        const carLength = 1.6;

        const spacing = 0.2;

        for (let i = 0; i < 15; i++) {

            const row = Math.floor(i / 5);

            const col = i % 5;

            const carX = parkingXStart + col * (carWidth + spacing);

            const carZ = parkingZStart + row * (carLength + spacing);

            const carGeo = new THREE.BoxGeometry(carWidth, 0.2, carLength);

            const carMat = new THREE.MeshStandardMaterial({ color: 0x88aadd, metalness: 0.7, roughness: 0.3 });

            const carMesh = new THREE.Mesh(carGeo, carMat);

            carMesh.position.set(carX, 0.1, carZ);

            carMesh.castShadow = true;

            scene.add(carMesh);

        }

        // Parking lot base

        const parkingLot = new THREE.Mesh(

            new THREE.BoxGeometry(6.5, 0.05, 7.5),

            new THREE.MeshStandardMaterial({ color: 0x4a5b6e, roughness: 0.8 })

        );

        parkingLot.position.set(-1.5, 0, 12.5);

        parkingLot.receiveShadow = true;

        scene.add(parkingLot);

        addFlatArea(-1.5, 12.5, 6.5, 7.5, 0x4a5b6e, "🚗 Outside Parking (15 cars)", 0.08);

        

        // --- Add some simple walls / boundaries? Optional but gives context ---

        // Put a surrounding low wall or just leave open.

        

        // Add a simple ground indicator for main building area

        const buildingBase = new THREE.Mesh(

            new THREE.PlaneGeometry(28, 22),

            new THREE.MeshStandardMaterial({ color: 0x2c3e4e, side: THREE.DoubleSide, transparent: true, opacity: 0.2 })

        );

        buildingBase.rotation.x = -Math.PI/2;

        buildingBase.position.y = -0.03;

        buildingBase.receiveShadow = false;

        scene.add(buildingBase);

        

        // --- Decorative elements: trees or simple columns ---

        const treeMat = new THREE.MeshStandardMaterial({ color: 0x5c9e5e });

        const trunkMat = new THREE.MeshStandardMaterial({ color: 0x8b5a2b });

        function addTree(x, z) {

            const trunk = new THREE.Mesh(new THREE.CylinderGeometry(0.3, 0.4, 0.8, 6), trunkMat);

            trunk.position.set(x, 0.4, z);

            trunk.castShadow = true;

            const foliage = new THREE.Mesh(new THREE.ConeGeometry(0.6, 0.9, 8), treeMat);

            foliage.position.set(x, 0.9, z);

            foliage.castShadow = true;

            scene.add(trunk);

            scene.add(foliage);

        }

        addTree(7, 13);

        addTree(9.5, 13);

        addTree(12, 13);

        addTree(-8, 12);

        addTree(-10, 9);

        

        // --- Simple animated helper to orient camera initially ---

        setTimeout(() => {

            camera.position.set(20, 14, 18);

            controls.target.set(2, 1, 2);

            controls.update();

        }, 100);

        

        // --- Animation Loop ---

        function animate() {

            requestAnimationFrame(animate);

            controls.update(); // update if user interacts

            renderer.render(scene, camera);

            labelRenderer.render(scene, camera);

        }

        animate();

        

        // Handle window resize

        window.addEventListener('resize', onWindowResize, false);

        function onWindowResize() {

            camera.aspect = window.innerWidth / window.innerHeight;

            camera.updateProjectionMatrix();

            renderer.setSize(window.innerWidth, window.innerHeight);

            labelRenderer.setSize(window.innerWidth, window.innerHeight);

        }

        

        console.log('3D facility ready');

    </script>
